---
rule-id: AV1710
title: Don't repeat the name of a class or enumeration in its members
severity: 1
---
class Employee
	{
		// Wrong!
		static GetEmployee() {...}
		DeleteEmployee() {...}
		
		// Right
		static Get() {...}
		Delete() {...}
		
		// Also correct.
		AddNewJob() {...}
		RegisterForMeeting() {...}
	}
